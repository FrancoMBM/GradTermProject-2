year <- function(date.column){
  
  year.factor <- c()
  
  for (i in 1:length(date.column)){
    
    if(date.column[i]  >= '2020-01-01'){
      
      year.factor[i] = '2020'
      
    }else if (date.column[i]  >= '2019-01-01') {
      year.factor[i] = '2019'
    }else if (date.column[i]  >= '2018-01-01') {
      year.factor[i] = '2018'
    }else{
      year.factor[i] = '2017'
    }
    
  }
  
  return(year.factor)
}
